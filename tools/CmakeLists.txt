CMAKE_MINIMUM_REQUIRED(VERSION 3.0)
PROJECT(BOW_BUILDER)
ADD_DEFINITIONS(-std=c++11)
SET(BLINDFIND_SOURCE_DIR "${PROJECT_SOURCE_DIR}/../")
SET(THIRD_PARTY_DIR "${BLINDFIND_SOURCE_DIR}/3rdParty")
SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${THIRD_PARTY_DIR}/DBoW3/lib/cmake" "${THIRD_PARTY_DIR}/opencv/share/OpenCV")
FIND_PACKAGE(OpenCV REQUIRED)
#FIND_PACKAGE(DBoW3 REQUIRED)
SET(SRC_FILE "${PROJECT_SOURCE_DIR}/index_builder.cpp" "${PROJECT_SOURCE_DIR}/DataBaseBuild.cpp")
SET(INCLUDE_DIR "${BLINDFIND_SOURCE_DIR}/3rdParty/DBoW3/include/DBoW3" "${BLINDFIND_SOURCE_DIR}/include" ${OpenCV_INCLUDE_DIRS})
SET(LIBRARY_DIR "${BLINDFIND_SOURCE_DIR}/3rdParty/DBoW3/lib")
SET(REQUIRED_LIB ${OpenCV_LIBS} ${DBoW3_LIBS})
INCLUDE_DIRECTORIES(${INCLUDE_DIR})
LINK_DIRECTORIES(${LIBRARY_DIR})
ADD_EXECUTABLE(BOW_BUILDER ${SRC_FILE})
TARGET_LINK_LIBRARIES(BOW_BUILDER ${REQUIRED_LIB} DBoW3)
